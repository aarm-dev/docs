export const BuilderRegistry = () => {
  const [search, setSearch] = useState("");
  const [sortField, setSortField] = useState("conformance");
  const [sortDir, setSortDir] = useState("asc");
  const [filterConf, setFilterConf] = useState("");
  const q = search.toLowerCase();
  const rows = BUILDERS.filter((b) => {
    if (q && !b.name.toLowerCase().includes(q) && !b.desc.toLowerCase().includes(q)) return false;
    if (filterConf && b.conformance !== filterConf) return false;
    return true;
  }).sort((a, b) => {
    let av, bv;
    if (sortField === "conformance") {
      av = a.conformance === "Conformant" ? 0 : 1;
      bv = b.conformance === "Conformant" ? 0 : 1;
    } else {
      av = a[sortField].toLowerCase();
      bv = b[sortField].toLowerCase();
    }
    if (av < bv) return sortDir === "asc" ? -1 : 1;
    if (av > bv) return sortDir === "asc" ? 1 : -1;
    return 0;
  });
  const doSort = (field) => {
    if (sortField === field) { setSortDir(sortDir === "asc" ? "desc" : "asc"); }
    else { setSortField(field); setSortDir("asc"); }
  };
const arrow = (field) => sortField === field ? (sortDir === "asc" ? " ▲" : " ▼") : " ⇅";
  const confN = BUILDERS.filter((b) => b.conformance === "Conformant").length;
  const alignN = BUILDERS.length - confN;
  return (
    <div className="not-prose" style={{ marginTop: "16px" }}>
      <div style={{ display: "flex", gap: "8px", marginBottom: "12px", flexWrap: "wrap", alignItems: "center" }}>
        <span style={{ background: "#f0fdf4", border: "1px solid #bbf7d0", borderRadius: "8px", padding: "4px 12px", fontSize: "13px" }}>
          <strong style={{ color: "#16a34a" }}>{confN}</strong>{" "}
          <span style={{ color: "#15803d" }}>Conformant</span>
        </span>
        <span style={{ background: "#eff6ff", border: "1px solid #bfdbfe", borderRadius: "8px", padding: "4px 12px", fontSize: "13px" }}>
          <strong style={{ color: "#2563eb" }}>{alignN}</strong>{" "}
          <span style={{ color: "#1d4ed8" }}>Aligned</span>
        </span>
        <span style={{ fontSize: "13px", color: "#94a3b8", marginLeft: "auto" }}>
          {rows.length} of {BUILDERS.length} shown
        </span>
      </div>
      <div style={{ display: "flex", gap: "8px", marginBottom: "12px", flexWrap: "wrap" }}>
        <input
          type="text"
          placeholder="Search builders..."
          value={search}
          onChange={(e) => setSearch(e.target.value)}
          style={{ flex: "1 1 200px", padding: "8px 12px", borderRadius: "8px", border: "1px solid #e2e8f0", fontSize: "13px", outline: "none", background: "transparent", color: "inherit", fontFamily: "inherit" }}
        />
        <select
          value={filterConf}
          onChange={(e) => setFilterConf(e.target.value)}
          style={{ padding: "8px 12px", borderRadius: "8px", border: "1px solid #e2e8f0", fontSize: "13px", background: "transparent", color: "inherit", cursor: "pointer", fontFamily: "inherit" }}
        >
          <option value="">All Status</option>
          <option value="Conformant">Conformant</option>
          <option value="Aligned">Aligned</option>
        </select>
        {(search || filterConf) ? (
          <button
            onClick={() => { setSearch(""); setFilterConf(""); }}
            style={{ padding: "8px 14px", borderRadius: "8px", border: "1px solid #fca5a5", fontSize: "13px", background: "#fef2f2", color: "#dc2626", cursor: "pointer", fontWeight: "600", fontFamily: "inherit" }}
          >
            Clear
          </button>
        ) : null}
      </div>
      <div style={{ overflowX: "auto", borderRadius: "10px", border: "1px solid #e2e8f0" }}>
        <table style={{ width: "100%", borderCollapse: "separate", borderSpacing: 0, fontSize: "13px", tableLayout: "auto" }}>
          <thead>
            <tr style={{ background: "rgba(0,0,0,0.02)" }}>
              <th onClick={() => doSort("name")} style={{ padding: "10px 14px 10px 20px", textAlign: "left", fontWeight: "600", color: "#475569", cursor: "pointer", userSelect: "none", borderBottom: "2px solid #e2e8f0", fontSize: "11px", textTransform: "uppercase", letterSpacing: "0.05em", whiteSpace: "nowrap" }}>
                Company<span style={{ color: sortField === "name" ? "#0ea5e9" : "#cbd5e1", fontSize: "10px" }}>{arrow("name")}</span>
              </th>
              <th onClick={() => doSort("conformance")} style={{ padding: "10px 14px", textAlign: "left", fontWeight: "600", color: "#475569", cursor: "pointer", userSelect: "none", borderBottom: "2px solid #e2e8f0", fontSize: "11px", textTransform: "uppercase", letterSpacing: "0.05em", whiteSpace: "nowrap", width: "110px" }}>
                Status<span style={{ color: sortField === "conformance" ? "#0ea5e9" : "#cbd5e1", fontSize: "10px" }}>{arrow("conformance")}</span>
              </th>
              <th style={{ padding: "10px 14px", textAlign: "left", fontWeight: "600", color: "#475569", borderBottom: "2px solid #e2e8f0", fontSize: "11px", textTransform: "uppercase", letterSpacing: "0.05em" }}>
                Description
              </th>
            </tr>
          </thead>
          <tbody>
            {rows.length === 0 ? (
              <tr>
                <td colSpan={3} style={{ padding: "48px 14px", textAlign: "center", color: "#94a3b8" }}>
                  No builders match your search.
                </td>
              </tr>
            ) : rows.map((b, i) => {
              const isConf = b.conformance === "Conformant";
              const domain = b.url.replace("https://", "").replace("http://", "").replace("www.", "").split("/")[0];
              return (
                <tr
                  key={b.name}
                  onClick={() => window.open(b.url, "_blank")}
                  style={{ cursor: "pointer", background: isConf ? "rgba(34,197,94,0.06)" : (i % 2 === 0 ? "transparent" : "rgba(0,0,0,0.01)") }}
                >
                  <td style={{ padding: "8px 14px 8px 20px", borderBottom: "1px solid #f1f5f9", minWidth: "200px" }}>
                    <div style={{ display: "flex", alignItems: "center", gap: "10px", flexWrap: "nowrap" }}>
                      <img
                        src={"https://www.google.com/s2/favicons?domain=" + domain + "&sz=64"}
                        alt=""
                        style={{ borderRadius: "4px", flexShrink: 0, width: "20px", height: "20px", minWidth: "20px", maxWidth: "20px", display: "block" }}
                      />
                      <span style={{ fontWeight: "600" }}>{b.name}</span>
                      <span style={{ color: "#cbd5e1", fontSize: "11px" }}>{"↗"}</span>
                    </div>
                  </td>
                  <td style={{ padding: "8px 14px", borderBottom: "1px solid #f1f5f9" }}>
                    <span style={{ display: "inline-block", padding: "2px 10px", borderRadius: "20px", fontSize: "10px", fontWeight: "700", letterSpacing: "0.03em", textTransform: "uppercase", color: "#fff", background: isConf ? "#22c55e" : "#3b82f6" }}>
                      {b.conformance}
                    </span>
                  </td>
                  <td style={{ padding: "8px 14px", borderBottom: "1px solid #f1f5f9", color: "#64748b", fontSize: "12px", lineHeight: "1.5" }}>
                    {b.desc}
                  </td>
                </tr>
              );
            })}
          </tbody>
        </table>
      </div>
      <div style={{ marginTop: "10px", fontSize: "11px", color: "#94a3b8" }}>
        Click any row to visit. Sort by clicking column headers.
      </div>
    </div>
  );
};

## How to Get Listed

### AARM Conformant

1. Satisfy all MUST requirements (R1-R7) as defined in the [Conformance Requirements](/conformance/requirements)
2. Complete the [Testing Protocol](/conformance/testing) and provide evidence
3. Wait 14 days for the conformance report if you passed the review

### AARM Aligned

1. Your product addresses one or more aspects of the AI runtime security problem
2. Submit a brief description of what you're building
3. No conformance testing required
